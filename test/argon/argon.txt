# Liquid Argon Example

# Species Definitions
Species 'Argon'
  # Atoms
  Atom    1   Ar     0.000    0.000    0.000    'Ar'

  # Isotopologues
  Isotopologue  'Ar36'  Ar=36
EndSpecies

# Define Configuration
Configuration 'liquid'
  Multiplier  10000
  Density  0.021300  atoms/A3
  Species  'Argon'  1.0
  Module AtomShake
  EndModule
EndConfiguration

# Samples
# Experimental data taken from Table 1 of Yarnell, Katz, Wenzel, and Koenig, Phys. Rev. A 7, 2130 (1973).
# Liquid Ar36 at 85 K. Data are normalised to <b2>, and oscillate around 1.0.
Sample  'Yarnell-85K'
  ReferenceData  'yarnell.sq'
  Module Partials
    Save true
    Configuration  'liquid'
    Weights neutron
    Isotopologue  'liquid'  'Argon'  'Ar36'  1.0
    StructureFactor  on
    QDepBroadening  0.02    # For SLS
    #NormalisedToAvSq
    #SubtractSelf
  EndModule
EndSample

# Pair Potentials
PairPotentials
  Generate  'LJ'  'Ar'  'Ar'
EndPairPotentials

Simulation
  MaxIterations 5
EndSimulation
