# Liquid Argon Example

# Species Definitions
Species 'Argon'
  # Atoms
  Atom    1   Ar     0.000    0.000    0.000    'Ar'

  # Isotopologues
  Isotopologue  'Ar36'  Ar=36
EndSpecies

# Define Configuration
Configuration 'liquid'
  Multiplier  10000
  Density  0.021300  atoms/A3
  SpeciesInfo  'Argon'  1.0
  UseOutputAsInput
  Module Energy
  EndModule
  Module AtomShake
  EndModule
  Module Energy
  EndModule
EndConfiguration

Module Partials 'Yarnell-85K'
  Save true
  Configuration  'liquid'
  Weights neutron
  Isotopologue  'liquid'  'Argon'  'Ar36'  1.0
  StructureFactor  on
  QDepBroadening  0.02    # For SLS
  Averaging  5
  AveragingScheme  Exponential
EndModule

# Experimental data taken from Table 1 of Yarnell, Katz, Wenzel, and Koenig, Phys. Rev. A 7, 2130 (1973).
# Liquid Ar36 at 85 K. Data are normalised to <b2>, and oscillate around 1.0.
Data  'Yarnell-85K'
  File  'data/yarnell.sq'
  NeutronNormalisation  AverageSquared
  SubtractAverageLevel  10.0
  AssociatedTo  'Yarnell-85K'
EndData

Module Test
  # Define targets
  Data  'Yarnell-85K'
  Frequency 5
EndModule

# Pair Potentials
PairPotentials
  Generate  'LJ'  'Ar'  'Ar'
EndPairPotentials

Simulation
  MaxIterations 50
EndSimulation
